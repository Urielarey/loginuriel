# Reglas de Firestore para permitir escritura sin autenticación
# Copia y pega estas reglas en Firebase Console > Firestore > Reglas

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permitir lectura y escritura a la colección login_attempts sin autenticación
    match /login_attempts/{document} {
      allow read, write: if true;
    }
    
    // Permitir lectura y escritura a cualquier documento (solo para desarrollo)
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
